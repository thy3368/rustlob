## 1. 五级流程模型概述

```
银行业五级流程模型层次:

┌─────────────────────────────────────────────────────────────────────────────┐
│ Level 1: 价值链 (Value Chain)                                               │
│ 定义: 企业核心价值活动领域                                                    │
│ 示例: 跨境支付服务                                                           │
├─────────────────────────────────────────────────────────────────────────────┤
│ Level 2: 流程域 (Process Area)                                              │
│ 定义: 价值链下的主要业务流程领域                                              │
│ 示例: 汇出汇款、汇入汇款、换汇、清结算                                        │
├─────────────────────────────────────────────────────────────────────────────┤
│ Level 3: 流程组 (Process Group)                                             │
│ 定义: 流程域内的端到端业务流程                                                │
│ 示例: B2B付款流程、C2C汇款流程                                               │
├─────────────────────────────────────────────────────────────────────────────┤
│ Level 4: 业务任务 (Task)                                                    │
│ 定义: 可独立执行的原子业务活动                                                │
│ 示例: 创建订单、风控审核、换汇执行                                            │
├─────────────────────────────────────────────────────────────────────────────┤
│ Level 5: 步骤与规则 (Step & Rule)                                           │
│ 定义: 任务执行的具体步骤和业务规则                                            │
│ 示例: 校验金额 > 0、查询实时汇率、计算手续费                                  │
└─────────────────────────────────────────────────────────────────────────────┘
```

|------|---------|-----------|---------|

| 核心业务 | L2.2 | 汇入汇款域 | 跨境收款、入账、退汇、追索 |
| 核心业务 | L2.3 | 换汇域 | 汇率报价、换汇成交、头寸对冲 |

```rust
// ============================================================================
// Command Handler Trait
// ============================================================================

#[async_trait]
pub trait CommandHandler<C: Command> {
    type Output;
    type Error;

    async fn handle(&self, command: C) -> Result<Self::Output, Self::Error>;
}

// ============================================================================
// CreatePaymentOrderHandler
// ============================================================================

pub struct CreatePaymentOrderHandler {
    order_repo: Arc<dyn OrderRepository>,
    merchant_service: Arc<dyn MerchantService>,
    fee_calculator: Arc<dyn FeeCalculator>,
    event_publisher: Arc<dyn EventPublisher>,
    idempotency_store: Arc<dyn IdempotencyStore>,
}

#[async_trait]
impl CommandHandler<CreatePaymentOrderCommand> for CreatePaymentOrderHandler {
    type Output = CreatePaymentOrderResult;
    type Error = OrderError;

    async fn handle(&self, cmd: CreatePaymentOrderCommand) -> Result<Self::Output, Self::Error> {
        // 1. 幂等性检查
        if let Some(existing) = self.idempotency_store
            .get(&cmd.idempotency_key).await? {
            return Ok(existing);
        }

        // 2. 验证商户状态
        let merchant = self.merchant_service
            .get_active_merchant(&cmd.merchant_id).await?;

        // 3. 验证走廊支持
        self.validate_corridor(&cmd.source, &cmd.destination)?;

        // 4. 验证限额
        self.validate_limits(&merchant, &cmd.amount).await?;

        // 5. 计算费用
        let fee = self.fee_calculator.calculate(
            &merchant,
            &cmd.order_type,
            &cmd.amount,
        )?;

        // 6. 创建订单
        let order = PaymentOrder::create(
            OrderId::generate(),
            cmd.merchant_id,
            cmd.order_type,
            cmd.source,
            cmd.destination,
            cmd.amount,
            fee,
        )?;

        // 7. 持久化
        self.order_repo.save(&order).await?;

        // 8. 发布事件
        let event = PaymentOrderCreatedEvent {
            order_id: order.id.clone(),
            merchant_id: order.merchant_id.clone(),
            amount: order.amount.clone(),
            created_at: Utc::now(),
        };
        self.event_publisher.publish(event).await?;

        // 9. 存储幂等性结果
        let result = CreatePaymentOrderResult {
            order_id: order.id,
            status: order.status,
            fee,
        };
        self.idempotency_store.set(&cmd.idempotency_key, &result).await?;

        Ok(result)
    }
}

// ExecuteFxDealHandler
// ============================================================================

pub struct ExecuteFxDealHandler {
    quote_repo: Arc<dyn FxQuoteRepository>,
    deal_repo: Arc<dyn FxDealRepository>,
    position_service: Arc<dyn PositionService>,
    account_service: Arc<dyn AccountService>,
    event_publisher: Arc<dyn EventPublisher>,
}

#[async_trait]
impl CommandHandler<ExecuteFxDealCommand> for ExecuteFxDealHandler {
    type Output = ExecuteFxDealResult;
    type Error = FxError;

    async fn handle(&self, cmd: ExecuteFxDealCommand) -> Result<Self::Output, Self::Error> {
        // 1. 获取并验证报价
        let quote = self.quote_repo
            .get_locked_quote(&cmd.quote_id).await?
            .ok_or(FxError::QuoteNotFound)?;

        // 验证报价有效性
        if quote.is_expired() {
            return Err(FxError::QuoteExpired);
        }
        if quote.customer_rate != cmd.confirmed_rate {
            return Err(FxError::RateMismatch);
        }

        // 2. 检查头寸限额
        self.position_service
            .validate_position_limit(&quote.dest_currency, quote.dest_amount)
            .await?;

        // 3. 开启事务执行换汇
        let deal = self.execute_fx_transaction(&quote).await?;

        // 4. 更新头寸
        self.position_service.update_position(
            &quote.source_currency,
            quote.source_amount,
            PositionDirection::Long,
        ).await?;

        self.position_service.update_position(
            &quote.dest_currency,
            quote.dest_amount,
            PositionDirection::Short,
        ).await?;

        // 5. 检查对冲触发
        if self.position_service.needs_hedging(&quote.dest_currency).await? {
            self.trigger_hedge(&quote.dest_currency).await?;
        }

        // 6. 发布事件
        let event = FxDealExecutedEvent {
            deal_id: deal.id.clone(),
            order_id: cmd.order_id,
            source_currency: quote.source_currency,
            dest_currency: quote.dest_currency,
            source_amount: quote.source_amount,
            dest_amount: quote.dest_amount,
            customer_rate: quote.customer_rate,
            dealt_at: Utc::now(),
        };
        self.event_publisher.publish(event).await?;

        Ok(ExecuteFxDealResult {
            deal_id: deal.id,
            dest_amount: deal.dest_amount,
        })
    }
}

// SubmitToChannelHandler
// ============================================================================

pub struct SubmitToChannelHandler {
    order_repo: Arc<dyn OrderRepository>,
    channel_gateway: Arc<dyn ChannelGateway>,
    circuit_breaker: Arc<CircuitBreaker>,
    event_publisher: Arc<dyn EventPublisher>,
}

#[async_trait]
impl CommandHandler<SubmitToChannelCommand> for SubmitToChannelHandler {
    type Output = SubmitToChannelResult;
    type Error = ChannelError;

    async fn handle(&self, cmd: SubmitToChannelCommand) -> Result<Self::Output, Self::Error> {
        // 1. 检查熔断器状态
        if !self.circuit_breaker.can_execute(&cmd.channel_id) {
            return Err(ChannelError::CircuitOpen);
        }

        // 2. 获取订单
        let mut order = self.order_repo
            .get(&cmd.order_id).await?
            .ok_or(ChannelError::OrderNotFound)?;

        // 3. 组装报文
        let message = self.channel_gateway
            .build_message(&cmd.payment_instruction, &cmd.channel_id)
            .await?;

        // 4. 签名
        let signed_message = self.channel_gateway
            .sign_message(message, &cmd.channel_id)
            .await?;

        // 5. 发送到通道
        let response = match self.channel_gateway
            .submit(&cmd.channel_id, signed_message)
            .await {
            Ok(resp) => {
                self.circuit_breaker.record_success(&cmd.channel_id);
                resp
            }
            Err(e) => {
                self.circuit_breaker.record_failure(&cmd.channel_id);
                return Err(e);
            }
        };

        // 6. 更新订单状态
        order.mark_channel_submitted(
            cmd.channel_id.clone(),
            response.channel_ref.clone(),
        )?;
        self.order_repo.save(&order).await?;

        // 7. 发布事件
        let event = ChannelSubmittedEvent {
            order_id: cmd.order_id,
            channel_id: cmd.channel_id,
            channel_ref: response.channel_ref.clone(),
            submitted_at: Utc::now(),
        };
        self.event_publisher.publish(event).await?;

        Ok(SubmitToChannelResult {
            channel_ref: response.channel_ref,
            estimated_completion: response.estimated_completion,
        })
    }
}
```

```rust
pub struct QueryFxRateHandler {
    lp_gateway: Arc<dyn LiquidityProviderGateway>,
    rate_cache: Arc<dyn RateCache>,
    event_publisher: Arc<dyn EventPublisher>,
}

#[async_trait]
impl CommandHandler<QueryFxRateCommand> for QueryFxRateHandler {
    type Output = FxRateInquiryResult;
    type Error = FxError;

    async fn handle(&self, cmd: QueryFxRateCommand) -> Result<Self::Output, Self::Error> {
        // 1. 检查缓存是否有有效报价
        if let Some(cached) = self.rate_cache
            .get(&cmd.source_currency, &cmd.dest_currency).await? {
            if cached.is_valid() {
                return Ok(cached);
            }
        }

        // 2. 并发查询所有LP
        let lps = if cmd.liquidity_providers.is_empty() {
            self.lp_gateway.get_active_lps(&cmd.source_currency, &cmd.dest_currency).await?
        } else {
            cmd.liquidity_providers.clone()
        };

        let quotes = self.lp_gateway
            .query_rates_parallel(&lps, &cmd.source_currency, &cmd.dest_currency, cmd.amount)
            .await?;

        // 3. 生成询价ID
        let inquiry_id = InquiryId::generate();

        // 4. 发布事件
        self.event_publisher.publish(FxRateQueriedEvent {
            inquiry_id: inquiry_id.clone(),
            source_currency: cmd.source_currency.clone(),
            dest_currency: cmd.dest_currency.clone(),
            lp_count: quotes.len(),
            timestamp: Utc::now(),
        }).await?;

        Ok(FxRateInquiryResult {
            inquiry_id,
            quotes,
            timestamp: Utc::now(),
        })
    }
}

pub struct ExecuteFxDealHandler {
    quote_store: Arc<dyn QuoteStore>,
    deal_repo: Arc<dyn FxDealRepository>,
    position_service: Arc<dyn PositionService>,
    lp_gateway: Arc<dyn LiquidityProviderGateway>,
    event_publisher: Arc<dyn EventPublisher>,
    idempotency_store: Arc<dyn IdempotencyStore>,
}

#[async_trait]
impl CommandHandler<ExecuteFxDealCommand> for ExecuteFxDealHandler {
    type Output = FxDealResult;
    type Error = FxError;

    async fn handle(&self, cmd: ExecuteFxDealCommand) -> Result<Self::Output, Self::Error> {
        // 1. 幂等性检查
        if let Some(existing) = self.idempotency_store
            .get(&cmd.idempotency_key).await? {
            return Ok(existing);
        }

        // 2. 验证报价有效性
        let quote = self.quote_store.get(&cmd.quote_id).await?
            .ok_or(FxError::QuoteNotFound)?;

        if quote.is_expired() {
            return Err(FxError::QuoteExpired);
        }

        // 3. 验证汇率一致性
        if quote.rate != cmd.confirmed_rate {
            return Err(FxError::RateMismatch);
        }

        // 4. 执行成交
        let deal = FxDeal::new(
            DealId::generate(),
            cmd.order_id.clone(),
            quote.currency_pair.clone(),
            cmd.confirmed_rate,
            cmd.source_amount,
            cmd.dest_amount,
            cmd.value_date,
        );

        // 5. 持久化
        self.deal_repo.save(&deal).await?;

        // 6. 更新头寸
        self.position_service.update_position(UpdatePositionCommand {
            command_id: CommandId::generate(),
            deal_id: deal.id.clone(),
            currency_pair: quote.currency_pair.clone(),
            position_change: cmd.dest_amount,
            change_type: PositionChangeType::CustomerDeal,
            timestamp: Utc::now(),
        }).await?;

        // 7. 发布事件
        self.event_publisher.publish(FxDealExecutedEvent {
            deal_id: deal.id.clone(),
            order_id: cmd.order_id.clone(),
            rate: cmd.confirmed_rate,
            source_amount: cmd.source_amount,
            dest_amount: cmd.dest_amount,
            timestamp: Utc::now(),
        }).await?;

        // 8. 保存幂等性结果
        let result = FxDealResult {
            deal_id: deal.id,
            status: DealStatus::Executed,
            executed_rate: cmd.confirmed_rate,
            source_amount: cmd.source_amount,
            dest_amount: cmd.dest_amount,
            value_date: cmd.value_date,
        };

        self.idempotency_store.set(&cmd.idempotency_key, &result).await?;

        Ok(result)
    }
}

pub struct ExecuteHedgeHandler {
    position_repo: Arc<dyn PositionRepository>,
    lp_gateway: Arc<dyn LiquidityProviderGateway>,
    hedge_repo: Arc<dyn HedgeRepository>,
    event_publisher: Arc<dyn EventPublisher>,
}

#[async_trait]
impl CommandHandler<ExecuteHedgeCommand> for ExecuteHedgeHandler {
    type Output = HedgeExecutionResult;
    type Error = FxError;

    async fn handle(&self, cmd: ExecuteHedgeCommand) -> Result<Self::Output, Self::Error> {
        // 1. 验证当前头寸
        let position = self.position_repo
            .get_current(&cmd.currency_pair).await?;

        // 2. 根据执行风格选择策略
        let execution_result = match cmd.execution_style {
            ExecutionStyle::Aggressive => {
                self.lp_gateway.execute_market_order(
                    &cmd.target_lps,
                    &cmd.currency_pair,
                    cmd.hedge_amount,
                    cmd.direction.clone(),
                ).await?
            }
            ExecutionStyle::Passive => {
                self.lp_gateway.place_limit_order(
                    &cmd.target_lps,
                    &cmd.currency_pair,
                    cmd.hedge_amount,
                    cmd.direction.clone(),
                    cmd.price_limit.ok_or(FxError::PriceLimitRequired)?,
                ).await?
            }
            ExecutionStyle::Twap => {
                self.lp_gateway.execute_twap(
                    &cmd.target_lps,
                    &cmd.currency_pair,
                    cmd.hedge_amount,
                    cmd.direction.clone(),
                    Duration::minutes(15),  // 15分钟TWAP
                ).await?
            }
            ExecutionStyle::Iceberg => {
                self.lp_gateway.execute_iceberg(
                    &cmd.target_lps,
                    &cmd.currency_pair,
                    cmd.hedge_amount,
                    cmd.direction.clone(),
                    cmd.hedge_amount / Decimal::from(10),  // 每次10%
                ).await?
            }
        };

        // 3. 记录对冲
        let hedge_record = HedgeRecord::new(
            HedgeId::generate(),
            cmd.hedge_request_id.clone(),
            cmd.currency_pair.clone(),
            cmd.hedge_amount,
            execution_result.executed_amount,
            execution_result.average_rate,
            HedgeStatus::Executed,
        );

        self.hedge_repo.save(&hedge_record).await?;

        // 4. 发布事件
        self.event_publisher.publish(HedgeExecutedEvent {
            hedge_id: hedge_record.id.clone(),
            currency_pair: cmd.currency_pair.clone(),
            requested_amount: cmd.hedge_amount,
            executed_amount: execution_result.executed_amount,
            average_rate: execution_result.average_rate,
            execution_style: cmd.execution_style.clone(),
            timestamp: Utc::now(),
        }).await?;

        Ok(HedgeExecutionResult {
            hedge_id: hedge_record.id,
            executed_amount: execution_result.executed_amount,
            average_rate: execution_result.average_rate,
            fill_ratio: execution_result.executed_amount / cmd.hedge_amount,
            status: HedgeStatus::Executed,
        })
    }
}
```

```rust
            .get(&cmd.discrepancy_id)
            .await?
            .ok_or(SettlementError::DiscrepancyNotFound)?;

        // 2. 根据处理动作执行
        let handling_result = match cmd.handling_action {
            DiscrepancyAction::AutoAdjust => {
                self.account_service.auto_adjust(&discrepancy).await?
            }
            DiscrepancyAction::ManualAdjust { adjustment_type, amount, account_id } => {
                self.account_service.manual_adjust(
                    &account_id,
                    &adjustment_type,
                    amount,
                    &cmd.operator_id,
                ).await?
            }

                self.chargeback_service.create(
                    &discrepancy,
                    reason,
                    deadline,
                ).await?

                    &cmd.discrepancy_id,
                    DiscrepancyStatus::PendingInvestigation,
                    Some(expected_resolution_date),
                ).await?
            }

                self.account_service.write_off(
                    &discrepancy,
                    &approval_id,
                    &cmd.operator_id,
                ).await?
            }
            DiscrepancyAction::Ignore { reason } => {
                self.discrepancy_repo.update_status(
                    &cmd.discrepancy_id,
                    DiscrepancyStatus::Ignored,
                    None,
                ).await?
            }
        };

        // 3. 记录处理日志
        self.discrepancy_repo.add_handling_log(
            &cmd.discrepancy_id,
            &cmd.handling_action,
```

```rust
pub struct DepositHandler {
    wallet_repo: Arc<dyn WalletRepository>,
    transaction_repo: Arc<dyn TransactionRepository>,
    event_publisher: Arc<dyn EventPublisher>,
    idempotency_store: Arc<dyn IdempotencyStore>,
}

#[async_trait]
impl CommandHandler<DepositCommand> for DepositHandler {
    type Output = DepositResult;
    type Error = AccountError;

    async fn handle(&self, cmd: DepositCommand) -> Result<Self::Output, Self::Error> {
        // 1. 幂等性检查
        if let Some(existing) = self.idempotency_store
            .get(&cmd.idempotency_key).await? {
            return Ok(existing);
        }

        // 2. 验证钱包
        let wallet = self.wallet_repo.get(&cmd.wallet_id).await?
            .ok_or(AccountError::WalletNotFound)?;

        if wallet.currency != cmd.currency {
            return Err(AccountError::CurrencyMismatch);
        }

        if !wallet.is_active() {
            return Err(AccountError::WalletNotActive);
        }

        // 3. 生成交易流水
        let transaction = AccountTransaction {
            id: TransactionId::generate(),
            wallet_id: cmd.wallet_id.clone(),
            transaction_type: TransactionType::Deposit,
            amount: cmd.amount,
            currency: cmd.currency.clone(),
            balance_before: wallet.available_balance,
            balance_after: wallet.available_balance + cmd.amount,
            reference: cmd.reference.clone(),
            source: format!("{:?}", cmd.source),
            timestamp: Utc::now(),
        };

        // 4. 更新余额（原子操作）
        self.wallet_repo.credit(
            &cmd.wallet_id,
            cmd.amount,
            &transaction.id,
        ).await?;

        // 5. 保存交易记录
        self.transaction_repo.save(&transaction).await?;

        // 6. 发布事件
        self.event_publisher.publish(DepositCompletedEvent {
            transaction_id: transaction.id.clone(),
            wallet_id: cmd.wallet_id.clone(),
            amount: cmd.amount,
            currency: cmd.currency.clone(),
            new_balance: transaction.balance_after,
            timestamp: Utc::now(),
        }).await?;

        // 7. 保存幂等性结果
        let result = DepositResult {
            transaction_id: transaction.id,
            new_balance: transaction.balance_after,
        };

        self.idempotency_store.set(&cmd.idempotency_key, &result).await?;

        Ok(result)
    }
}

pub struct FreezeBalanceHandler {
    wallet_repo: Arc<dyn WalletRepository>,
    freeze_repo: Arc<dyn FreezeRepository>,
    event_publisher: Arc<dyn EventPublisher>,
}

#[async_trait]
impl CommandHandler<FreezeBalanceCommand> for FreezeBalanceHandler {
    type Output = FreezeResult;
    type Error = AccountError;

    async fn handle(&self, cmd: FreezeBalanceCommand) -> Result<Self::Output, Self::Error> {
        // 1. 验证钱包余额
        let wallet = self.wallet_repo.get(&cmd.wallet_id).await?
            .ok_or(AccountError::WalletNotFound)?;

        if wallet.available_balance < cmd.amount {
            return Err(AccountError::InsufficientBalance);

        // 2. 创建冻结记录
        let freeze = FreezeRecord {
            id: FreezeId::generate(),
            wallet_id: cmd.wallet_id.clone(),
            amount: cmd.amount,
            reason: cmd.freeze_reason.clone(),
            reference_type: cmd.reference_type.clone(),
            reference_id: cmd.reference_id.clone(),
            status: FreezeStatus::Active,
            created_at: Utc::now(),
            expiry: cmd.expiry,
        };

        // 3. 原子操作：减少可用余额，增加冻结余额
        self.wallet_repo.freeze(&cmd.wallet_id, cmd.amount).await?;

        // 4. 保存冻结记录
        self.freeze_repo.save(&freeze).await?;

        // 5. 发布事件
        self.event_publisher.publish(BalanceFrozenEvent {
            freeze_id: freeze.id.clone(),
            wallet_id: cmd.wallet_id.clone(),
            amount: cmd.amount,
            reason: cmd.freeze_reason.clone(),
            reference_id: cmd.reference_id.clone(),
            timestamp: Utc::now(),
        }).await?;

        Ok(FreezeResult {
            freeze_id: freeze.id,
            frozen_amount: cmd.amount,
            remaining_available: wallet.available_balance - cmd.amount,
        })
    }
}

pub struct TransferHandler {
    wallet_repo: Arc<dyn WalletRepository>,
    transaction_repo: Arc<dyn TransactionRepository>,
    event_publisher: Arc<dyn EventPublisher>,
    idempotency_store: Arc<dyn IdempotencyStore>,
}

#[async_trait]
impl CommandHandler<TransferCommand> for TransferHandler {
    type Output = TransferResult;
    type Error = AccountError;

    async fn handle(&self, cmd: TransferCommand) -> Result<Self::Output, Self::Error> {
        // 1. 幂等性检查
        if let Some(existing) = self.idempotency_store
            .get(&cmd.idempotency_key).await? {
            return Ok(existing);
        }

        // 2. 验证源钱包
        let from_wallet = self.wallet_repo.get(&cmd.from_wallet_id).await?
            .ok_or(AccountError::WalletNotFound)?;

        if from_wallet.available_balance < cmd.amount {
            return Err(AccountError::InsufficientBalance);
        }

        if from_wallet.currency != cmd.currency {
            return Err(AccountError::CurrencyMismatch);
        }

        // 3. 验证目标钱包
        let to_wallet = self.wallet_repo.get(&cmd.to_wallet_id).await?
            .ok_or(AccountError::WalletNotFound)?;

        if to_wallet.currency != cmd.currency {
            return Err(AccountError::CurrencyMismatch);
        }

        // 4. 执行转账（事务）
        let transfer_id = TransferId::generate();
        self.wallet_repo.transfer(
            &cmd.from_wallet_id,
            &cmd.to_wallet_id,
            cmd.amount,
            &transfer_id,
        ).await?;

        // 5. 保存交易记录（双边）
        let debit_txn = AccountTransaction {
            id: TransactionId::generate(),
            wallet_id: cmd.from_wallet_id.clone(),
            transaction_type: TransactionType::TransferOut,
            amount: -cmd.amount,
            currency: cmd.currency.clone(),
            balance_before: from_wallet.available_balance,
            balance_after: from_wallet.available_balance - cmd.amount,
            reference: cmd.reference.clone(),
            source: format!("Transfer to {}", cmd.to_wallet_id),
            timestamp: Utc::now(),
        };

        let credit_txn = AccountTransaction {
            id: TransactionId::generate(),
            wallet_id: cmd.to_wallet_id.clone(),
            transaction_type: TransactionType::TransferIn,
            amount: cmd.amount,
            currency: cmd.currency.clone(),
            balance_before: to_wallet.available_balance,
            balance_after: to_wallet.available_balance + cmd.amount,
            reference: cmd.reference.clone(),
            source: format!("Transfer from {}", cmd.from_wallet_id),
            timestamp: Utc::now(),
        };

        self.transaction_repo.save_batch(&[debit_txn, credit_txn]).await?;

        // 6. 发布事件
        self.event_publisher.publish(TransferCompletedEvent {
            transfer_id: transfer_id.clone(),
            from_wallet_id: cmd.from_wallet_id.clone(),
            to_wallet_id: cmd.to_wallet_id.clone(),
            amount: cmd.amount,
            currency: cmd.currency.clone(),
            transfer_type: cmd.transfer_type.clone(),
            timestamp: Utc::now(),
        }).await?;

        // 7. 保存幂等性结果
        let result = TransferResult {
            transfer_id,
            from_new_balance: from_wallet.available_balance - cmd.amount,
            to_new_balance: to_wallet.available_balance + cmd.amount,
        };

        self.idempotency_store.set(&cmd.idempotency_key, &result).await?;

        Ok(result)
    }
}
```

```
│ │                                                                          │ │
```

### 6.2 状态转换规则

```rust
// 状态机实现
pub struct OrderStateMachine;

impl OrderStateMachine {
    pub fn can_transition(from: OrderStatus, to: OrderStatus) -> bool {
        match (from, to) {
            // 创建后的转换
            (Created, PaymentPending) => true,
            (Created, Cancelled) => true,

            // 待付款的转换
            (PaymentPending, PaymentReceived) => true,
            (PaymentPending, Cancelled) => true,

            // 已收款的转换
            (PaymentReceived, ComplianceCheck) => true,

            // 合规审核的转换
            (ComplianceCheck, FxProcessing) => true,      // 通过
            (ComplianceCheck, ComplianceHold) => true,    // 暂停
            (ComplianceCheck, Failed) => true,            // 拒绝

            // 合规暂停的转换
            (ComplianceHold, ComplianceCheck) => true,    // 重新审核
            (ComplianceHold, Cancelled) => true,          // 超时取消

            // 换汇处理的转换
            (FxProcessing, Routing) => true,
            (FxProcessing, Failed) => true,

            // 路由的转换
            (Routing, ChannelSubmitted) => true,
            (Routing, Failed) => true,

            // 已提交通道的转换
            (ChannelSubmitted, ChannelProcessing) => true,
            (ChannelSubmitted, Failed) => true,

            // 通道处理中的转换
            (ChannelProcessing, Completed) => true,
            (ChannelProcessing, Failed) => true,

            // 失败的转换
            (Failed, Refunding) => true,

            // 退款中的转换
            (Refunding, Refunded) => true,

            _ => false,
        }
    }

    pub fn transition(order: &mut Order, to: OrderStatus) -> Result<(), StateError> {
        if !Self::can_transition(order.status, to) {
            return Err(StateError::InvalidTransition {
                from: order.status,
                to,
            });
        }

        // 记录状态变更
        order.status_history.push(StatusChange {
            from: order.status,
            to,
            changed_at: Utc::now(),
            reason: None,
        });

        order.status = to;
        order.updated_at = Utc::now();

        Ok(())
    }
}
```
